1.求半年内的时间
this.endDate = new Date();
let curDate = this.endDate.getTime();
let halfYear = (365 / 2) * 24 * 3600 * 1000;
this.startDate = new Date(curDate - halfYear);

  /[1-9]\d{3}(.{0,1}\d{2}){0,1}/

  /^\d{6}$/
